##straw ammoniation######
import numpy as np

# Given values
fertilizer_lakh_tonnes = 163.33766  # in lakh tonnes
fertilizer_tg = fertilizer_lakh_tonnes / 10  # Convert to Tg
fertilizer_unc = 0.10  # 10%

percent_straw = 0.05  # 5%
percent_straw_unc = 0.10  # 10%

n_content = 0.467  # 46.7%
n_content_unc = 0  # assume fixed

# === Method 1: Uncertainty Propagation ===
central_value = fertilizer_tg * percent_straw * n_content

# Relative uncertainty propagation
rel_unc_total = np.sqrt(fertilizer_unc**2 + percent_straw_unc**2)
abs_unc = central_value * rel_unc_total

print("=== Straw Ammoniation N Input (Propagation) ===")
print(f"Value: {central_value:.4f} Tg N")
print(f"Uncertainty: ±{abs_unc:.4f} Tg N")
print(f"Relative Uncertainty: ±{rel_unc_total*100:.2f}%")


##food and kitchen waste######

import numpy as np

# Input values
food_waste = 79_000_000  # kg/capita/year
population = 487_702_168  # people
n_content = 0.01  # 1% nitrogen content

# Relative uncertainties (as fractions)
ru_food_waste = 0.10
ru_population = 0.05
ru_n_content = 0.10

# Total nitrogen in kg/year
total_n_kg = food_waste * population * n_content

# Convert to Tg (1 Tg = 1e12 g = 1e9 kg)
total_n_tg = total_n_kg / 1e9

# Propagation of relative uncertainty (multiplicative formula)
combined_ru = np.sqrt(ru_food_waste**2 + ru_population**2 + ru_n_content**2)

# Absolute uncertainty in Tg
uncertainty_tg = total_n_tg * combined_ru

# Output
print(f"Estimated Total N from food waste: {total_n_tg:.4f} Tg N/year ± {uncertainty_tg:.4f} Tg")
print(f"Relative Uncertainty: ±{combined_ru * 100:.2f}%")


###fish powder######

import numpy as np

# Values
aquatic_production = 0.7
human_consumption = 0.2
product_export = 0.04

# Relative uncertainties (in decimals)
ru_aq = 0.10
ru_hc = 0.10
ru_exp = 0.10

# Step 1: Calculate fish powder
fish_powder = aquatic_production - human_consumption - product_export

# Step 2: Absolute uncertainties
abs_unc_aq = aquatic_production * ru_aq
abs_unc_hc = human_consumption * ru_hc
abs_unc_exp = product_export * ru_exp

# Step 3: Combine uncertainties (add in quadrature for subtraction)
total_unc = np.sqrt(abs_unc_aq**2 + abs_unc_hc**2 + abs_unc_exp**2)

# Step 4: Relative uncertainty (%)
rel_unc_percent = (total_unc / fish_powder) * 100 if fish_powder != 0 else np.nan

# Output
print(f"Fish Powder Estimate: {fish_powder:.4f}")
print(f"Uncertainty: ±{total_unc:.4f}")
print(f"Relative Uncertainty: ±{rel_unc_percent:.2f}%")

##change in breeding stock=accumulation

import pandas as pd
import numpy as np

# ==== Step 1: Input data ====
data = {
    "animal": ["chicken", "duck", "swine", "cow", "buffalo", "goat", "sheep"],
    "population": [314269000, 35507000, 885211, 51568474, 109719011, 150248487, 68099762],
    "slaughtering": [2839116000, 35626000, 9346277.91, 36504854.76, 11523861.85, 55547065.12, 23433066.08],
    "products": [4472690, 46313.49, 383199.36, 4414240, 1968157, 661697, 328615.28],  # tonnes
    "n_content": [3.09, 2.48, 2.11, 3.18, 3.18, 3.04, 3.04],  # %
}

# Relative uncertainties
ru_population = 0.20
ru_slaughtering = 0.10
ru_products = 0.10
ru_n_content = 0.10

# ==== Step 2: Create DataFrame ====
df = pd.DataFrame(data)

# ==== Step 3: Calculate cutability ====
df["cutability"] = np.where(
    df["animal"].isin(["chicken", "duck"]),
    df["products"] / df["slaughtering"],  # same units (tonnes)
    (df["products"] * 1e3) / df["slaughtering"]  # convert to kg/animal for others
)

# ==== Step 4: Breeding stock calculation ====
df["breeding_stock"] = (df["population"] * df["cutability"] * df["n_content"]) / 100  # /100 to scale % to Tg N

# ==== Step 5: Propagate relative uncertainty ====
df["ru_total"] = np.sqrt(
    ru_population**2 +
    ru_slaughtering**2 +
    ru_products**2 +
    ru_n_content**2
)

# ==== Step 6: Absolute uncertainty and CI ====
df["abs_uncertainty"] = df["breeding_stock"] * df["ru_total"]
df["lower"] = df["breeding_stock"] - df["abs_uncertainty"]
df["upper"] = df["breeding_stock"] + df["abs_uncertainty"]

# ==== Step 7: Total breeding stock (convert kg to Tg) ====
total_sum_kg = df["breeding_stock"].sum()
total_uncertainty_kg = np.sqrt(np.sum(df["abs_uncertainty"]**2))  # RSS

# Convert to Tg
total_sum = total_sum_kg / 1e9
total_uncertainty = total_uncertainty_kg / 1e9
total_lower = total_sum - total_uncertainty
total_upper = total_sum + total_uncertainty
total_ru = total_uncertainty / total_sum * 100

# ==== Step 8: Output ====
print("=== Breeding Stock by Animal ===")
df[["breeding_stock", "abs_uncertainty", "lower", "upper"]] = df[["breeding_stock", "abs_uncertainty", "lower", "upper"]] / 1e12
print(df[["animal", "breeding_stock", "abs_uncertainty", "lower", "upper"]].round(6))

print("\n=== TOTAL BREEDING STOCK N ===")
print(f"Sum: {total_sum:.6f} Tg N")
print(f"Uncertainty (±1σ): ±{total_uncertainty:.6f} Tg N")
print(f"95% Confidence Interval: {total_lower:.6f} – {total_upper:.6f} Tg N")
print(f"Relative Uncertainty: ±{total_ru:.2f}%")

import numpy as np
import pandas as pd

# Animal categories
animals = [
    "Assess", "Buffalo", "Camel", "Cattle, dairy", "Cattle, Non dairy",
    "Chickens, broilers", "Chickens, layers", "Ducks", "Goats", "Horses",
    "Mules and hinnies", "Sheep", "Swine, breeding", "Swine, market"
]

# Population and excretion factor data
population = np.array([
    253500, 109719011, 223810, 51568474, 142913881,
    476763000, 314269000, 35507000, 150248487, 340000,
    82984, 68099762, 885211, 7966900
], dtype=np.float64)

excretion_factor = np.array([
    92.34, 45, 55, 60, 40,
    0.73, 0.59, 0.81, 9, 45,
    92.3, 19.21, 11, 11
], dtype=np.float64)

# Uncertainties (10%)
pop_uncertainty = 0.20 * population  ## IPCC(CHAPTER 10)
ef_uncertainty = 0.10 * excretion_factor

# Manure Produced = Population × Excretion Factor (kg)
manure_produced_kg = population * excretion_factor
manure_produced_Tg = manure_produced_kg / 1e9

# Uncertainty propagation
relative_uncertainty_squared = (pop_uncertainty / population) ** 2 + (ef_uncertainty / excretion_factor) ** 2
manure_uncertainty_kg = manure_produced_kg * np.sqrt(relative_uncertainty_squared)
manure_uncertainty_Tg = manure_uncertainty_kg / 1e9

# Total manure and total uncertainty
total_manure_Tg = manure_produced_Tg.sum()
total_uncertainty_Tg = np.sqrt(np.sum(manure_uncertainty_Tg**2))

# Partial uncertainty contribution
total_uncertainty_squared = np.sum(manure_uncertainty_Tg ** 2)
partial_uncertainties = (manure_uncertainty_Tg ** 2 / total_uncertainty_squared) * 100

# DataFrame
df = pd.DataFrame({
    "Animal": animals,
    "Population": population,
    "Excretion Factor": excretion_factor,
    "Manure Produced (Tg)": manure_produced_Tg,
    "Uncertainty (± Tg)": manure_uncertainty_Tg,
    "Partial Uncertainty (%)": partial_uncertainties
})

# Sort by partial uncertainty
df_sorted = df.sort_values(by="Partial Uncertainty (%)", ascending=False).reset_index(drop=True)

# Display sorted table
print(df_sorted.to_string(index=False))

# Summary output
print("\n--- Summary ---")
print(f"Total manure produced: {total_manure_Tg:.5f} Tg")
print(f"Total uncertainty in manure produced: ±{total_uncertainty_Tg:.5f} Tg")

], dtype=np.float64)

    92.34, 45, 55, 60, 40,
    0.73, 0.59, 0.81, 9, 45,
    92.3, 19.21, 11, 11

ef_uncertainty = 0.10 * excretion_factor

manure_produced_Tg = manure_produced_kg / 1e9
rel_uncertainty_sq = (pop_uncertainty / population) ** 2 + (ef_uncertainty / excretion_factor) ** 2

# Grazing percentages

    "Cattle, Non dairy": 0.22,
    "Buffalo": 0.19,
    "Cattle, dairy": 0.27,
    "Goats": 0.95,

    "Mules and hinnies": 0.95,

    grazing_percent_dict.get(animal, 0.0) for animal in animals
], dtype=np.float64)

rel_grazing_uncertainty_sq = rel_uncertainty_sq + np.where(
    grazing_percentages > 0,
    (grazing_percent_uncertainty / grazing_percentages) ** 2,
    0
)
grazing_manure_uncertainty_Tg = grazing_manure_Tg * np.sqrt(rel_grazing_uncertainty_sq)

import pandas as pd
import numpy as np

def weighted_mean_and_uncertainty(c_vals, x_vals, std_vals):
    """
    Calculates the weighted mean and propagated uncertainty (std dev)
    c_vals: list of counts [C1, C2, C3]
    x_vals: list of percentages [x1, x2, x3]
    std_vals: list of std deviations [σ1, σ2, σ3]
    """
    total_weight = sum(c_vals)
    if total_weight == 0:
        return np.nan, np.nan
    
    weighted_mean = sum(c * x for c, x in zip(c_vals, x_vals)) / total_weight
    weighted_variance = sum((c**2) * (s**2) for c, s in zip(c_vals, std_vals)) / (total_weight**2)
    weighted_std = np.sqrt(weighted_variance)
    return weighted_mean, weighted_std

}

total_inputs = np.array(total_inputs)
mean_input = np.mean(total_inputs)
std_input = np.std(total_inputs)
pi_lower, pi_upper = np.percentile(total_inputs, [2.5, 97.5])
ru_percent = (std_input / mean_input) * 100

# Print results
print(f"✅ Mean Total N Input = {mean_input:.3f} Tg")
print(f"✅ Standard Deviation = {std_input:.3f} Tg")
print(f"✅ 95% Prediction Interval = [{pi_lower:.3f}, {pi_upper:.3f}] Tg")
print(f"✅ Relative Uncertainty = ±{ru_percent:.2f}%")

# Plotting the distribution
plt.figure(figsize=(10, 6))
plt.hist(total_inputs, bins=50, density=True, alpha=0.7, edgecolor='black', label='Monte Carlo Results')

# Overlay normal distribution
x = np.linspace(mean_input - 4 * std_input, mean_input + 4 * std_input, 1000)
pdf = (1 / (std_input * np.sqrt(2 * np.pi))) * np.exp(-0.5 * ((x - mean_input) / std_input) ** 2)
plt.plot(x, pdf, 'r-', label='Normal Distribution')

# Annotate
plt.xlabel("Total N Input (Tg)")
plt.ylabel("Density")
plt.title("Monte Carlo Simulation: Sum of N Inputs")
plt.legend()

# Add text box with summary stats
plt.text(x.min(), plt.ylim()[1]*0.9,
         f"Mean = {mean_input:.3f} Tg\n"
         f"SD = {std_input:.3f}\n"
         f"95% PI = [{pi_lower:.3f}, {pi_upper:.3f}]\n"
         f"RU = ±{ru_percent:.2f}%",
         fontsize=11, verticalalignment='top', bbox=dict(facecolor='white', alpha=0.8))

# Save figure
plt.tight_layout()
plt.savefig("Total_N_Inputs_MonteCarlo.png", dpi=1200)
plt.show()
